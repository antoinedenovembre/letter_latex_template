% Date formatting
\newcommand{\monthname}{% \monthname
	\ifcase\month\or
		January\or February\or March\or April\or May\or June\or
		July\or August\or September\or October\or November\or December
	\fi
}%

% Data formatting in french
\newcommand{\monthnameFR}{% \monthnameFR
	\ifcase\month\or
		Janvier\or Février\or Mars\or Avril\or Mai\or Juin\or
		Juillet\or Août\or Septembre\or Octobre\or Novembre\or Décembre
	\fi
}%

% Today command but in french
\newcommand{\todayFR}{\number\day~\monthnameFR~\number\year}
